# Inputs
part_identifier = input("Enter the part number: ")
order_quantity = int(input("Enter the quantity: "))

# Logic for unit cost based on part number
if part_identifier == "10" or part_identifier == "55":
    cost_per_unit = 1.00
elif part_identifier == "99":
    cost_per_unit = 2.00
elif part_identifier == "80" or part_identifier == "70":
    cost_per_unit = 3.00
else:
    cost_per_unit = 5.00

# Calculation
total_order_cost = order_quantity * cost_per_unit

# Output
print(f"Part Number: {part_identifier}")
print(f"Unit Cost: ${cost_per_unit:.2f}")
print(f"Total Cost: ${total_order_cost:.2f}")
