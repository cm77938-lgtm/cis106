def compute_assessed_value(county_name, market_value):
    if county_name == "Cook":
        percentage = 0.90
    elif county_name == "DuPage":
        percentage = 0.80
    elif county_name == "McHenry":
        percentage = 0.75
    elif county_name == "Kane":
        percentage = 0.60
    else:
        percentage = 0.70
    return market_value * percentage

sum_market = 0
sum_assessed = 0
user_choice = "Yes"
while user_choice == "Yes":
    county_val = input("Enter county: ")
    market_val = float(input("Enter market value: "))
    
    assessed_val = compute_assessed_value(county_val, market_val)
    sum_market += market_val
    sum_assessed += assessed_val
    
    print("Assessed value:", assessed_val)
    user_choice = input("Continue (Yes/No)? ")

print("Total market values:", sum_market)
print("Total assessed values:", sum_assessed)
