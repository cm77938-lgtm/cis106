def compute_ticket_price(miles_distance):
    if miles_distance >= 30:
        return 12
    elif miles_distance >= 20:
        return 10
    elif miles_distance >= 10:
        return 8
    else:
        return 5

running_total = 0

    ticket_cost = compute_ticket_price(miles_input)
