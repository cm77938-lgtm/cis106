# --- Dynamic List Operations ---

# Prompt for the number of items and create the list [cite: 4, 5, 6]
total_count = int(input("How many items will be in the list? "))
items_list = []

for index in range(total_count):
    entry_val = int(input("Enter an integer: "))
    items_list.append(entry_val)

# Display the initial list [cite: 7]
print("Initial list:", items_list)

# Insert 99 at position 1 and display 
items_list.insert(1, 99)
print("After inserting 99:", items_list)

# Replace the value at position 1 with 100 and display [cite: 9]
items_list[1] = 100
print("After replacing with 100:", items_list)

# Create and display a second list [cite: 10]
extra_list = [500, 600, 700, 800, 900]
print("Second list:", extra_list)

# Extend the first list with the second and display 
items_list.extend(extra_list)
print("Extended list:", items_list)

# Remove specific value 800 and display 
items_list.remove(800)
print("After removing 800:", items_list)

# Remove the third item (index 2) and display 
items_list.pop(2)
print("After removing third item:", items_list)
