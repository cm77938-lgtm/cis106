def compute_final_price(vehicle_make, vehicle_model, ev_status, sticker_price):
    if ev_status == "Y":
        discount = 0.30
    elif vehicle_make == "Honda" and vehicle_model == "Accord":
        discount = 0.10
    elif vehicle_make == "Toyota" and vehicle_model == "Rav4":
        discount = 0.15
    else:
        discount = 0.05
        
    discounted_price = sticker_price - (sticker_price * discount)
    final_price = discounted_price * 1.07
    return final_price

total_msrp = 0
total_sales = 0
user_choice = "Yes"
while user_choice == "Yes":
    make_val = input("Enter make: ")
    model_val = input("Enter model: ")
    ev_code_val = input("Is it electric (Y/N)? ")
    price_val = float(input("Enter MSRP: "))
    
    final_price = compute_final_price(make_val, model_val, ev_code_val, price_val)
    total_msrp += price_val
    total_sales += final_price
    
    print("Out the door price:", final_price)
    user_choice = input("Continue (Yes/No)? ")

print("Total MSRP sum:", total_msrp)
print("Total sales sum:", total_sales)
